Tipos de Variáveis

Como saber os tipos de variáveis com "typeof" .

// variáveis
const fruta = "Banana";
const idade = 40;
const preco = 3.50;
const levar = true;
let novoValor;

// typeof => exibe resultado no console
console.log("A variável fruta é: " + typeof fruta); // string
console.log("A variável preco é: " + typeof preco); // número
console.log("A variável idade é: " + typeof preco); // número
console.log("A variável levar é: " + typeof levar); // boolean
console.log("A variável novoValor é: " + typeof novoValor); //undefined

// perguntas
console.log("A variável preco é u núero inteiro? " + Number.isInteger(preco));
console.log("A variável idade é u núero inteiro? " + Number.isInteger(idade));
